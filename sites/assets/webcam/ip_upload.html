IP Upload Example
(Applet and Navigator Streaming support)

 

My current IP Address is 192.168.1.100

You can use our online wizard to create the html code.

<!-- visionGS Stream Code Begin -->

<SCRIPT language="JavaScript" type="text/javascript">
BN = navigator.appName;
if (BN == "Netscape")
{
document.write('<IMG SRC="http://192.168.1.100/video/jpeg-1" BORDER=0 WIDTH=320 HEIGHT=240>');
}
else
{
document.write('<APPLET CODEBASE="http://192.168.1.100/" CODE="VgsStream.class" WIDTH="320" HEIGHT="240 ">');
document.write('<PARAM NAME="port" VALUE="80">');
document.write('<PARAM NAME="username" VALUE="/">');
document.write('<PARAM NAME="password" VALUE="/">');
document.write('<PARAM NAME="device" VALUE="1">');
document.write('<\/APPLET>');
}
</SCRIPT>
